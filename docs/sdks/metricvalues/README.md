## list

```typescript
const statsig = new Statsig({
  statsigApiKey: "<YOUR_API_KEY_HERE>",
});

async function run() {
  const result = await statsig.metricValues.list({
    date: "<value>",
    limit: 10,
    page: 1,
  });

  // Handle the result
  console.log(result)
}

run();
```
